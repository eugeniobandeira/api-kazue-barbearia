CREATE TABLE `user` (
  `ID_USER` int NOT NULL AUTO_INCREMENT,
  `ID_STATUS` int NOT NULL,
  `DS_NAME` varchar(30) NOT NULL,
  `DS_SURNAME` varchar(30) NOT NULL,
  `DS_NICKNAME` varchar(30) DEFAULT NULL,
  `DS_EMAIL` varchar(30) NOT NULL,
  `DS_PHONE` varchar(30) NOT NULL,
  `DS_ROLE` varchar(20) NOT NULL DEFAULT 'CUSTOMER',
  `DT_BIRTH` date NOT NULL,
  `DT_REGISTRY` date NOT NULL,
  `DS_PASSWORD` longtext NOT NULL,
  PRIMARY KEY (`ID_USER`),
  KEY `ID_STATUS` (`ID_STATUS`),
  CONSTRAINT `user_ibfk_1` FOREIGN KEY (`ID_STATUS`) REFERENCES `status` (`ID_STATUS`)
) ENGINE=InnoDB AUTO_INCREMENT=15 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
