CREATE TABLE `tbl_barber` (
  `ID_BARBER` int NOT NULL AUTO_INCREMENT,
  `DS_NAME` varchar(30) NOT NULL,
  `DS_SURNAME` varchar(30) NOT NULL,
  `DS_NICKNAME` varchar(30) DEFAULT NULL,
  `DS_EMAIL` varchar(30) NOT NULL,
  `DS_PHONE` varchar(30) NOT NULL,
  `DS_ROLE` varchar(20) NOT NULL DEFAULT 'ADMIN',
  `TP_STATUS` char(1) NOT NULL,
  `DT_BIRTH` date NOT NULL,
  `DT_REGISTRY` date NOT NULL,
  `DS_PASSWORD` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`ID_BARBER`)
) ENGINE=InnoDB AUTO_INCREMENT=8 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
