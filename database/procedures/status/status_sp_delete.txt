CREATE DEFINER=`root`@`localhost` PROCEDURE `STATUS_SP_DELETE`(
    IN P_ID_STATUS INT
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        RESIGNAL;
    END;

    START TRANSACTION;

    DELETE FROM STATUS 
    WHERE 
        ID_STATUS = P_ID_STATUS;

    COMMIT;
END //

DELIMITER ;
