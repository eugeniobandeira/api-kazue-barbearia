CREATE DEFINER=`root`@`localhost` PROCEDURE `STATUS_SP_GET_BY_ID_OR_DESCRIPTION`(
    IN P_ID_STATUS INT,          
    IN P_DS_STATUS VARCHAR(20)  
)
BEGIN
    SELECT 
        ID_STATUS, 
        DS_STATUS 
    FROM 
        STATUS
    WHERE 
        (P_ID_STATUS IS NULL OR ID_STATUS = P_ID_STATUS) AND
        (P_DS_STATUS IS NULL OR DS_STATUS = P_DS_STATUS);
END //
