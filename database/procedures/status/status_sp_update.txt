CREATE DEFINER=`root`@`localhost` PROCEDURE `STATUS_SP_UPDATE`(
	IN P_ID_STATUS INT,
	IN P_DS_STATUS VARCHAR(30)
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        RESIGNAL;
    END;

    START TRANSACTION;
    
    UPDATE STATUS 
    SET DS_STATUS = P_DS_STATUS
    WHERE ID_STATUS = P_ID_STATUS;
    
    COMMIT;
END //

DELIMITER ;
