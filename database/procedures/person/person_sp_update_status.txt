CREATE DEFINER=`root`@`localhost` PROCEDURE `PERSON_SP_UPDATE_STATUS`(
    IN P_ID_PERSON INT,
    IN P_ID_STATUS INT
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        RESIGNAL;
    END;

    START TRANSACTION;

    UPDATE PERSON 
    SET 
        ID_STATUS = P_ID_STATUS
    WHERE 
        ID_PERSON = P_ID_PERSON;

    COMMIT;
END //

DELIMITER ;
