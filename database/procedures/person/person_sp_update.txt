DELIMITER //

CREATE DEFINER=`root`@`localhost` PROCEDURE `PERSON_SP_UPDATE`(
    IN P_ID_PERSON INT,
    IN P_DS_NAME VARCHAR(30),
    IN P_DS_SURNAME VARCHAR(30),
    IN P_DS_NICKNAME VARCHAR(30),
    IN P_DS_EMAIL VARCHAR(100),
    IN P_DS_PHONE VARCHAR(30)
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        RESIGNAL;
    END;

    START TRANSACTION;

    UPDATE PERSON 
    SET 
        DS_NAME = P_DS_NAME,
        DS_SURNAME = P_DS_SURNAME,
        DS_NICKNAME = P_DS_NICKNAME,
        DS_EMAIL = P_DS_EMAIL,
        DS_PHONE = P_DS_PHONE
    WHERE 
        ID_PERSON = P_ID_PERSON;

    COMMIT;
END //

DELIMITER ;
