CREATE DEFINER=`root`@`localhost` PROCEDURE `USER_SP_UPDATE_STATUS`(
    IN P_ID_USER INT,
    IN P_ID_STATUS INT
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        RESIGNAL;
    END;

    START TRANSACTION;

    UPDATE USER 
    SET 
        ID_STATUS = P_ID_STATUS
    WHERE 
        ID_USER = P_ID_USER;

    COMMIT;
END
