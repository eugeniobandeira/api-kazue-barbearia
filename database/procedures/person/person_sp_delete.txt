CREATE DEFINER=`root`@`localhost` PROCEDURE `PERSON_SP_DELETE`(
    IN P_ID_PERSON INT
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        RESIGNAL;
    END;

    START TRANSACTION;

    DELETE FROM PERSON 
    WHERE 
        ID_PERSON = P_ID_PERSON;

    COMMIT;
END //

DELIMITER ;
